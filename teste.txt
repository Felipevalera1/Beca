import datetime
from dateutil.relativedelta import relativedelta

def get_last_months(meses):
  lista_meses = []
  data_atual = datetime.datetime.now()
  for mes in range(1, meses+1):
    mes_passado = data_atual - relativedelta(months=mes)
    mes_passado = mes_passado.strftime('%Y%m%d')
    lista_meses.append(mes_passado)
  return lista_meses
